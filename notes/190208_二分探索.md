```cpp
    int head = lower_bound(a.begin(), a.end(), 5) - a.begin();
```

### ABC077 C Snuke Festival

#### 二分探索による解法

```cpp
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
using ll = long long;

int main() {
    int N; cin >> N;
    vector<int> a(N), b(N), c(N);
    for (int& ai : a) cin >> ai;
    for (int& bi : b) cin >> bi;
    for (int& ci : c) cin >> ci;
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    sort(c.begin(), c.end());
    ll ans = 0;
    for (int i = 0; i < N; i++) {
        int upper = lower_bound(a.begin(), a.end(), b[i]) - a.begin();
        int lower = c.end() - upper_bound(c.begin(), c.end(), b[i]);
        ans += (ll)(upper) * (lower);
    }
    cout << ans << endl;
    return 0;
}
```

#### （参考）尺取り法による解法

```cpp
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
using ll = long long;

int main() {
    int N; cin >> N;
    vector<int> a(N), b(N), c(N);
    for (int& ai : a) cin >> ai;
    for (int& bi : b) cin >> bi;
    for (int& ci : c) cin >> ci;
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    sort(c.begin(), c.end());
    ll ans = 0;
	int it1 = 0, it3 = 0;
	for (int i = 0; i < N; i++) {
		while (it1 < N && a[it1] < b[i]) it1++;
		while (it3 < N && c[it3] <= b[i]) it3++;
		ans += (ll)it1 * (N - it3);
	}
    cout << ans << endl;
	return 0;
}
```

## 「解を決めて二分探索」

```cpp
#include <algorithm>
#include <iostream>
#include <vector>
#include <cassert>
using namespace std;
using ll = long long;

bool check(ll x) {
    return true;
}

ll binary_search(ll ok, ll ng) {
    assert(check(ok) == true);
    assert(check(ng) == false);
    while (abs(ng - ok) > 1) {
        ll mid = (ng + ok) / 2;
        (check(mid) ? ok : ng) = mid;
    }
    return ok;
}

int main() {
    ll ok = 100000 * 10000;
    ll ng = 0;
    cout << binary_search(ok, ng) << endl;
}
```

### ARC050 B 花束

```cpp
using ll = long long;
const ll INF = 1LL << 61;   // 2.3e18

ll R, B, x, y;

// K 個の花束を作ることができるか
bool check(ll K) {
    if (R - K < 0 || B - K < 0) return false;
    ll cr = (R - K) / (x - 1);
    ll cb = (B - K) / (y - 1);
    return cr + cb >= K;
}

ll binary_search(ll ok, ll ng) {}

int main() {
    cin >> R >> B >> x >> y;
    ll ng = INF;
    ll ok = 0;
    cout << binary_search(ok, ng) << endl;
    return 0;
}
```

### [AOJ 0600 Baumkuchen](https://onlinejudge.u-aizu.ac.jp/challenges/sources/JOI/Final/0600?year=2014)

```cpp
#include <algorithm>
#include <iostream>
#include <vector>
#include <cassert>
using namespace std;
using ll = long long;

int n;
vector<ll> a;
ll total;       // バームクーヘン１ホールのサイズ

// ３本とも X 以上にできるか判定
bool check(ll X) {
    for (int i = 0; i < n; i++) {   // 1つ目の切り込み
        auto tail = a.begin() + i + n + 1;
        int j = lower_bound(a.begin() + i, tail, a[i] + X) - a.begin();
        int k = lower_bound(a.begin() + j, tail, a[j] + X) - a.begin();
        if (a[j] - a[i] >= X && a[k] - a[j] >= X && a[i + n] - a[k] >= X) {
            return true;
        }
    }
    return false;
}

ll binary_search(ll ok, ll ng) {}

int main() {
    cin >> n; a.resize(2 * n + 1);  // 環状
    for (int i = 1; i <= n; i++) {
        cin >> a[i]; a[i + n] = a[i]; total += a[i];
    }
    for (int i = 1; i < 2 * n + 1; i++) {
        a[i] += a[i - 1];
    }
    ll ok = 0;
    ll ng = total / 3 + 1;
    cout << binary_search(ok, ng) << endl;
    return 0;
}
```
